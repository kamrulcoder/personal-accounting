    try {
        await mongoose.connect(process.env.MONGO_URI);
        app.listen(PORT, () => {
            console.log(`Server is running on port ${PORT}`);
        });
        console.log('MongoDB connected successfully');

        console.error('MongoDB connection error:', error);
        process.exit(1);
    }
}

export default connectDB;
